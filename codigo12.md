# Usamos bucles for para imprimir el dibujo

# Imprimir la parte ascendente del dibujo
for i in range(1, 6):
    print('*' * i)

# Imprimir la parte descendente del dibujo
for i in range(4, 0, -1):
    print('*' * i)
